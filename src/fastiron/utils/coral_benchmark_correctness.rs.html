14/// Runs additional tests according to the [BenchType].
15pub fn coral_benchmark_correctness<T: CustomFloat>(mcresults: &MonteCarloResults<T>) {
16    if mcresults.bench_type == BenchType::Standard {
17        return;
18    }
19
20    // only on mpi rank 0 in QS code
21    balance_ratio_test(mcresults);
22    balance_event_test(mcresults);
23    missing_particle_test(mcresults);
24
25    // add a condition on cycles/total particles?
26    fluence_test(mcresults);
27}

29/// Test Balance Tallies for relative correctness.
30///
31/// Expected ratios of absorbs, fissions, scatters are maintained
32/// withing some tolerance, based on input expectation.
33pub fn balance_ratio_test<T: CustomFloat>(mcresults: &MonteCarloResults<T>) {
34    println!("Testing if ratios for absorption, fission & scattering are maintained...");
35
36    let balance_tally = &mcresults.balance_cumulative;
37    let absorb: T = FromPrimitive::from_u64(balance_tally[TalliedEvent::Absorb]).unwrap();
38    let fission: T = FromPrimitive::from_u64(balance_tally[TalliedEvent::Fission]).unwrap();
39    let scatter: T = FromPrimitive::from_u64(balance_tally[TalliedEvent::Scatter]).unwrap();
40    let (fission_ratio, scatter_ratio, absorb_ratio, percent_tolerance): (T, T, T, T) =
41        if mcresults.bench_type == BenchType::Coral1 {
42            (
43                FromPrimitive::from_f64(0.05).unwrap(),
44                FromPrimitive::from_f64(1.0).unwrap(),
45                FromPrimitive::from_f64(0.04).unwrap(),
46                one(),
47            )
48        } else {
49            // then it's BenchType::Coral2
50            // can be verified from parsing & first if-branch of the calling function
51            (
52                FromPrimitive::from_f64(0.075).unwrap(),
53                FromPrimitive::from_f64(0.830).unwrap(),
54                FromPrimitive::from_f64(0.094).unwrap(),
55                FromPrimitive::from_f64(1.1).unwrap(),
56            )
57        };
58    let tolerance = percent_tolerance / FromPrimitive::from_f64(100.0).unwrap();
59
60    let abs2sct: T = ((absorb / absorb_ratio) * (scatter_ratio / scatter) - one()).abs();
61    let abs2fsn: T = ((absorb / absorb_ratio) * (fission_ratio / fission) - one()).abs();
62    let sct2abs: T = ((scatter / scatter_ratio) * (absorb_ratio / absorb) - one()).abs();
63    let sct2fsn: T = ((scatter / scatter_ratio) * (fission_ratio / fission) - one()).abs();
64    let fsn2abs: T = ((fission / fission_ratio) * (absorb_ratio / absorb) - one()).abs();
65    let fsn2sct: T = ((fission / fission_ratio) * (scatter_ratio / scatter) - one()).abs();
66
67    // pass if no ratio is above tolerance
68    let pass: bool = !((abs2sct > tolerance)
69        | (abs2fsn > tolerance)
70        | (sct2abs > tolerance)
71        | (sct2fsn > tolerance)
72        | (fsn2abs > tolerance)
73        | (fsn2sct > tolerance));
74    if pass {
75        println!("PASS:: Ratios are maintained within {percent_tolerance}%");
76    } else {
77        println!("FAIL:: Ratios are not maintained with {percent_tolerance}% tolerance");
78        println!("Total absorb: {absorb}");
79        println!("Total scatter: {scatter}");
80        println!("Total fission: {fission}");
81        println!("absorb to scatter: {abs2sct}");
82        println!("absorb to fission: {abs2fsn}");
83        println!("scatter to absorb: {sct2abs}");
84        println!("scatter to fission: {sct2fsn}");
85        println!("fission to absorb: {fsn2abs}");
86        println!("fission to scatter: {fsn2sct}");
87    }
88}

90/// Test Balance Tallies for equality in number of facet crossing
91/// and collision events.
92pub fn balance_event_test<T: CustomFloat>(mcresults: &MonteCarloResults<T>) {
93    println!("Testing balance between number of facet crossings and reactions...");
94
95    let balance_tally = &mcresults.balance_cumulative;
96    let facet_crossing: T = FromPrimitive::from_u64(
97        balance_tally[TalliedEvent::NumSegments]
98            - balance_tally[TalliedEvent::Collision]
99            - balance_tally[TalliedEvent::Census],
100    )
101    .unwrap();
102    let collision: T = FromPrimitive::from_u64(balance_tally[TalliedEvent::Collision]).unwrap();
103
104    let ratio: T = (facet_crossing / collision - one()).abs();
105    let percent_tolerance: T = one();
106
107    let pass: bool = ratio <= percent_tolerance / (FromPrimitive::from_f64(100.0).unwrap());
108    if pass {
109        println!("PASS:: Ratio maintained within {percent_tolerance}%");
110    } else {
111        println!("FAIL:: Ratio not maintained within {percent_tolerance}%");
112        println!("facet crossing to collision: {ratio}");
113    }
114}

116/// Test for lost particles during the simulation.
117///
118/// This test should always succeed unless test for
119/// done was broken, or we are running with 1 MPI rank
120/// and so never preform this test during test_for_done
121pub fn missing_particle_test<T: CustomFloat>(mcresults: &MonteCarloResults<T>) {
122    println!("Testing for lost / unaccounted for particles in this simulation...");
123
124    let bt = &mcresults.balance_cumulative;
125    let gains: u64 = bt[TalliedEvent::Start]
126        + bt[TalliedEvent::Source]
127        + bt[TalliedEvent::Produce]
128        + bt[TalliedEvent::Split];
129    let losses: u64 = bt[TalliedEvent::Absorb]
130        + bt[TalliedEvent::Census]
131        + bt[TalliedEvent::Escape]
132        + bt[TalliedEvent::OverRr]
133        + bt[TalliedEvent::WeightRr]
134        + bt[TalliedEvent::Fission];
135
136    if gains == losses {
137        println!("PASS:: No particles lost during run");
138    } else {
139        println!("FAIL:: Particles lost during run");
140    }
141}

143/// Test that the scalar flux is homogenous across cells for the problem.
144///
145/// This test really requires a slot of particles or cycles or both
146/// This solution should converge to a homogenous solution
147pub fn fluence_test<T: CustomFloat>(mcresults: &MonteCarloResults<T>) {
148    println!("Testing fluence for homogeneity across the cells");
149    let mut max_diff: T = zero();
150    let mut local_sum: T = zero();
151    mcresults
152        .fluence
153        .cell
154        .iter()
155        .for_each(|val| local_sum += *val);
156
157    let average: T = local_sum / FromPrimitive::from_usize(mcresults.fluence.cell.len()).unwrap();
158    mcresults.fluence.cell.iter().for_each(|cell_value| {
159        let percent_diff: T = (*cell_value - average).abs()
160            / ((*cell_value + average) / FromPrimitive::from_f64(2.0).unwrap())
161            * FromPrimitive::from_f64(100.0).unwrap();
162        max_diff = max_diff.max(percent_diff);
163    });
164    let percent_tolerance: T = FromPrimitive::from_f64(6.0).unwrap();
165    let pass = max_diff <= percent_tolerance;
166    if pass {
167        println!("PASS:: Fluence is homogeneous across cells within {percent_tolerance}%");
168    } else {
169        println!("FAIL:: Fluence is not homogeneous across cells within {percent_tolerance}%");
170        println!("Current max difference: {max_diff}%");
171        println!("Try running more particles / cycles to check if the max difference % goes down");
172    }
