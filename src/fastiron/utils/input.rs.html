139/// Updates the Parameters structure passed as argument using the
140/// provided input file. The file is first separated into blocks
141/// with the rsplit call. The blocks are then used to complete the
142/// parameter structure passed as argument.
143pub fn parse_input_file<T: CustomFloat>(
144    filename: String,
145    params: &mut Parameters<T>,
146) -> Result<(), Vec<InputError>> {
147    let mut content = String::new();
148
149    let mut file = match File::open(filename) {
150        Ok(file) => file,
151        Err(_) => return Err(vec![InputError::BadInputFile]),
152    };
153
154    file.read_to_string(&mut content).unwrap();
155
156    let res: Vec<Result<(), InputError>> = content
157        .rsplit("\n\n")
158        .map(|raw_block| {
159            if let Some(val) = raw_block.find('\n') {
160                let some_struct: Block = serde_yaml::from_str(&raw_block[val + 1..]).unwrap();
161                //println!("{:#?}", some_struct); // uncomment if a parsing issue occur.
162
163                match &raw_block[0..val] {
164                    "Simulation:" => params.update_simulation_parameters(some_struct),
165                    "Geometry:" => match GeometryParameters::from_block(some_struct) {
166                        Ok(some_geometry) => {
167                            params.add_geometry_parameter(some_geometry);
168                            return Ok(());
169                        }
170                        Err(e) => Err(e),
171                    },
172                    "Material:" => match MaterialParameters::from_block(some_struct) {
173                        Ok(some_material) => {
174                            params.add_material_parameter(some_material);
175                            return Ok(());
176                        }
177                        Err(e) => Err(e),
178                    },
179                    "CrossSection:" => match CrossSectionParameters::from_block(some_struct) {
180                        Ok(some_cross_section) => {
181                            params.add_cross_section_parameter(some_cross_section);
182                            return Ok(());
183                        }
184                        Err(e) => Err(e),
185                    },
186                    _ => Err(InputError::BadBlockType),
187                }?;
188            }
189            Ok(())
190        })
191        .collect();
192
193    let errors: Vec<InputError> = res.iter().filter_map(|r| r.clone().err()).collect();
194
195    if errors.is_empty() {
196        return Ok(());
197    }
198    Err(errors)
199}
