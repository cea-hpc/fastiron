10/// Enum used to represent a reaction type.
11#[derive(Debug, PartialEq, Clone, Copy)]
12pub enum ReactionType {
13    /// Value for a scattering reaction.
14    Scatter,
15    /// Value for an absorption reaction.
16    Absorption,
17    /// Value for a fission reaction.
18    Fission,
19}

21/// Structure used to represent a polynomial function.
22///
23/// Private fields represent the coefficients, `aa` corresponding to the leading
24/// coefficient , `ee` to the product of all roots.
25#[derive(Debug)]
26pub struct Polynomial<T: CustomFloat> {
27    /// Leading coefficient of the polynomial function.
28    pub aa: T,
29    /// Degree 3 coefficient of the polynomial function.
30    pub bb: T,
31    /// Degree 2 coefficient of the polynomial function.
32    pub cc: T,
33    /// Degree 1 coefficient of the polynomial function.
34    pub dd: T,
35    /// Degree 0 coefficient of the polynomial function.
36    pub ee: T,
37}
38
39impl<T: CustomFloat> Polynomial<T> {
40    /// Returns the value of the polynomial function in `xx`.
41    pub fn val(&self, xx: T) -> T {
42        self.ee + xx * (self.dd + xx * (self.cc + xx * (self.bb + xx * self.aa)))
43    }
44}

46/// Lowest-level structure to represent a reaction.
47///
48/// A reaction is characterized by three elements: a probability density
49/// (cross-section), the type of the reaction and a statistical offset
50/// (nu bar).
51#[derive(Debug)]
52pub struct NuclearDataReaction<T: CustomFloat> {
53    /// Cross-section of the reaction, i.e. the discretization of its
54    /// probability density function. Discretization is done according to
55    /// energy levels.
56    pub cross_section: Vec<T>,
57    /// Type of the reaction.
58    pub reaction_type: ReactionType,
59    /// Statistical offset. This value is used when sampling for the number
60    /// of offspring particles created by a reaction. See the
61    /// [`sample_collision()`][crate::particles::mc_particle::MCParticle::sample_collision()] method
62    /// for more information.
63    pub nu_bar: T,
64}
65
66impl<T: CustomFloat> NuclearDataReaction<T> {
67    /// Constructor.
68    pub fn new(
69        rtype: ReactionType,
70        nu_bar: T,
71        energies: &[T],
72        polynomial: &Polynomial<T>,
73        reaction_cross_section: T,
74    ) -> Self {
75        let n_groups = energies.len() - 1;
76        let mut xsection: Vec<f64> = vec![zero(); n_groups];
77        let mut normal_value: f64 = 0.0;
78        let two: T = FromPrimitive::from_f64(2.0).unwrap();
79
80        (0..n_groups).for_each(|ii| {
81            let energy: T = (energies[ii] + energies[ii + 1]) / two;
82            // 10^(Poly(log10(energy)))
83            // this here gives a value too big for f32
84            xsection[ii] = 10.0_f64.powf(polynomial.val(energy.log10()).to_f64().unwrap());
85
86            if (energies[ii + 1] >= T::one()) & (normal_value == 0.0) {
87                normal_value = xsection[ii];
88            }
89        });
90        let scale = reaction_cross_section.to_f64().unwrap() / normal_value;
91        (0..n_groups).for_each(|ii| {
92            xsection[ii] *= scale;
93        });
94
95        let xsection_t: Vec<T> = xsection
96            .iter()
97            .map(|ff| FromPrimitive::from_f64(*ff).unwrap())
98            .collect();
99
100        Self {
101            cross_section: xsection_t,
102            reaction_type: rtype,
103            nu_bar,
104        }
105    }
106}

108/// Structure used to hold a list of reactions.
109///
110/// The list of reaction held by the structure is specific to a particle species.
111/// At the current time there is only one type of particle in the simulation.
112#[derive(Debug, Default)]
113pub struct NuclearDataSpecies<T: CustomFloat> {
114    /// List of reactions.
115    pub reactions: Vec<NuclearDataReaction<T>>,
116}
117
118impl<T: CustomFloat> NuclearDataSpecies<T> {
119    /// Builds a reaction and adds it to the internal list.
120    pub fn add_reaction(
121        &mut self,
122        rtype: ReactionType,
123        nu_bar: T,
124        energies: &[T],
125        polynomial: &Polynomial<T>,
126        reaction_cross_section: T,
127    ) {
128        self.reactions.push(NuclearDataReaction::new(
129            rtype,
130            nu_bar,
131            energies,
132            polynomial,
133            reaction_cross_section,
134        ))
135    }
136}

144/// Top level structure used to handle all things related to
145/// nuclear data.
146#[derive(Debug, Default)]
147pub struct NuclearData<T: CustomFloat> {
148    /// Total number of energy groups.
149    pub num_energy_groups: usize,
150    /// Isotope-sorted reaction list.
151    pub isotopes: Vec<NuclearDataIsotope<T>>,
152    /// Discrete value of the energy spectrum.
153    pub energies: Vec<T>,
154}
155
156impl<T: CustomFloat> NuclearData<T> {
157    /// Constructor.
158    pub fn new(num_groups: usize, energy_low: T, energy_high: T) -> Self {
159        let mut energies = vec![zero(); num_groups + 1];
160        let length: T = FromPrimitive::from_usize(num_groups + 1).unwrap();
161        // complete energy levels
162        energies[0] = energy_low;
163        energies[num_groups] = energy_high;
164        let log_low: T = energy_low.ln();
165        let log_high: T = energy_high.ln();
166        let delta = (log_high - log_low) / length;
167
168        (1..num_groups).for_each(|ii| {
169            let step = FromPrimitive::from_usize(ii).unwrap();
170            let log_value: T = log_low + delta * step;
171            energies[ii] = log_value.exp();
172        });
173
174        Self {
175            num_energy_groups: num_groups,
176            isotopes: Vec::new(),
177            energies,
178        }
179    }
180
181    /// Adds an isotope to the internal list.
182    ///
183    /// The isotope is built from the material data and cross-sections
184    /// specified in the material.
185    pub fn add_isotope(
186        &mut self,
187        cross_section: &FxHashMap<String, Polynomial<T>>,
188        mp: &MaterialParameters<T>,
189        nu_bar: T,
190    ) -> usize {
191        // for readability purposes
192        let n_reactions: usize = mp.n_reactions;
193        let fission_function: &Polynomial<T> = &cross_section[&mp.fission_cross_section];
194        let scatter_function: &Polynomial<T> = &cross_section[&mp.scattering_cross_section];
195        let absorption_function: &Polynomial<T> = &cross_section[&mp.absorption_cross_section];
196        let total_cross_section: T = mp.total_cross_section;
197        let fission_weight: T = mp.fission_cross_section_ratio;
198        let scatter_weight: T = mp.scattering_cross_section_ratio;
199        let absorption_weight: T = mp.absorbtion_cross_section_ratio;
200
201        self.isotopes.push(vec![NuclearDataSpecies::default()]);
202        let total_weight = fission_weight + scatter_weight + absorption_weight;
203
204        let mut n_fission = n_reactions / 3;
205        let mut n_scatter = n_reactions / 3;
206        let n_absorption = n_reactions / 3;
207
208        // set reaction distribution
209        match n_reactions % 3 {
210            2 => {
211                n_fission += 1;
212                n_scatter += 1;
213            }
214            1 => {
215                n_scatter += 1;
216            }
217            0 => (),
218            _ => unreachable!(),
219        }
220        let mut f: T = FromPrimitive::from_usize(n_fission).unwrap();
221        let fission_xsection: T = (total_cross_section * fission_weight) / (f * total_weight);
222        f = FromPrimitive::from_usize(n_scatter).unwrap();
223        let scatter_xsection: T = (total_cross_section * scatter_weight) / (f * total_weight);
224        f = FromPrimitive::from_usize(n_absorption).unwrap();
225        let absorption_xsection: T = (total_cross_section * absorption_weight) / (f * total_weight);
226
227        let n = self.isotopes.len();
228        self.isotopes[n - 1][0].reactions.reserve(n_reactions);
229
230        (0..n_reactions).for_each(|ii| match ii % 3 {
231            0 => self.isotopes[n - 1][0].add_reaction(
232                ReactionType::Scatter,
233                nu_bar,
234                &self.energies,
235                scatter_function,
236                scatter_xsection,
237            ),
238            1 => self.isotopes[n - 1][0].add_reaction(
239                ReactionType::Fission,
240                nu_bar,
241                &self.energies,
242                fission_function,
243                fission_xsection,
244            ),
245            2 => self.isotopes[n - 1][0].add_reaction(
246                ReactionType::Absorption,
247                nu_bar,
248                &self.energies,
249                absorption_function,
250                absorption_xsection,
251            ),
252            _ => unreachable!(),
253        });
254        self.isotopes.len() - 1
255    }
256
257    /// Returns the energy group a specific energy belongs to by using a dichotomy search.
258    pub fn get_energy_groups(&self, energy: T) -> usize {
259        let num_energies = self.energies.len();
260
261        // extreme low
262        if energy <= self.energies[0] {
263            return 0;
264        }
265        // extreme high
266        if energy > self.energies[num_energies - 1] {
267            return num_energies - 1;
268        }
269
270        // dichotomy search
271        let mut high = num_energies - 1;
272        let mut low: usize = 0;
273
274        while high != low + 1 {
275            let mid = (high + low) / 2;
276            if energy < self.energies[mid] {
277                high = mid;
278            } else {
279                low = mid;
280            }
281        }
282
283        low
284    }
285
286    /// Returns the total cross-section for a given energy group.
287    pub fn get_total_cross_section(&self, isotope_index: usize, group: usize) -> T {
288        // sum all reaction's xsection for a given isotope at a given energy level
289        self.isotopes[isotope_index][0]
290            .reactions
291            .iter()
292            .map(|reaction| reaction.cross_section[group])
293            .sum()
294    }
