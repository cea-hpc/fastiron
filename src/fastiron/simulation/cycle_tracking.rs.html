26/// Main steps of the `CycleTracking` section.
27///
28/// The particle at the specified index is loaded, tracked and updated accordingly.
29/// Depending on the outcome of the tracking, it is either set as processed or
30/// invalidated.
31pub fn cycle_tracking_guts<T: CustomFloat>(
32    mcdata: &MonteCarloData<T>,
33    mcunit: &MonteCarloUnit<T>,
34    particle: &mut MCParticle<T>,
35    balance: &mut Balance,
36    extra: &mut ParticleCollection<T>,
37) {
38    // set age & time to census
39    if particle.time_to_census <= zero() {
40        particle.time_to_census += mcdata.params.simulation_params.dt;
41    }
42    particle.age = particle.age.max(zero());
43    // update energy & task
44    particle.energy_group = mcdata
45        .nuclear_data
46        .get_energy_groups(particle.kinetic_energy);
47
48    cycle_tracking_function(mcdata, mcunit, particle, balance, extra);
49}
50
51fn cycle_tracking_function<T: CustomFloat>(
52    mcdata: &MonteCarloData<T>,
53    mcunit: &MonteCarloUnit<T>,
54    particle: &mut MCParticle<T>,
55    balance: &mut Balance,
56    extra: &mut ParticleCollection<T>,
57) {
58    let mut keep_tracking: bool;
59
60    loop {
61        // compute event for segment
62        let segment_outcome = outcome(mcdata, mcunit, particle);
63        // update # of segments
64        balance[TalliedEvent::NumSegments] += 1;
65        particle.num_segments += one();
66        // update scalar flux tally
67        mcunit.tallies.scalar_flux_domain[(particle.cell, particle.energy_group)]
68            .fetch_update(Ordering::Relaxed, Ordering::Relaxed, |x| {
69                Some(x + particle.segment_path_length * particle.weight)
70            })
71            .unwrap();
72
73        match segment_outcome {
74            MCSegmentOutcome::Collision => {
75                let mat_gid = mcunit.domain.cell_state[particle.cell].material;
76                let cell_nb_density = mcunit.domain.cell_state[particle.cell].cell_number_density;
77
78                keep_tracking =
79                    collision_event(mcdata, balance, mat_gid, cell_nb_density, particle, extra);
80
81                particle.energy_group = mcdata
82                    .nuclear_data
83                    .get_energy_groups(particle.kinetic_energy);
84                if !keep_tracking {
85                    particle.species = Species::Unknown;
86                }
87            }
88            MCSegmentOutcome::FacetCrossing => {
89                // crossed facet data
90                let facet_adjacency = &mcunit.domain.mesh.cell_connectivity[particle.cell].facet
91                    [particle.facet]
92                    .subfacet;
93
94                facet_crossing_event(particle, facet_adjacency);
95
96                keep_tracking = match particle.last_event {
97                    // ~~~ on unit case
98                    // on-unit transit
99                    MCTallyEvent::FacetCrossingTransitExit => true,
100                    // bound reflection
101                    MCTallyEvent::FacetCrossingReflection => {
102                        // plane on which particle is reflected
103                        let plane =
104                            &mcunit.domain.mesh.cell_geometry[particle.cell][particle.facet];
105
106                        reflect_particle(particle, plane);
107                        true
108                    }
109                    // ~~~ off unit case
110                    // off-unit transit
111                    MCTallyEvent::FacetCrossingCommunication => {
112                        unimplemented!()
113                    }
114                    // bound escape
115                    MCTallyEvent::FacetCrossingEscape => {
116                        balance[TalliedEvent::Escape] += 1;
117                        particle.last_event = MCTallyEvent::FacetCrossingEscape;
118                        particle.species = Species::Unknown;
119                        false
120                    }
121                    // ~~~ other enum values are for collision & census, not facet crossing
122                    _ => unreachable!(),
123                };
124            }
125            MCSegmentOutcome::Census => {
126                balance[TalliedEvent::Census] += 1;
127                // we're done tracking the particle FOR THIS STEP; Species stays valid
128                keep_tracking = false;
129            }
130        }
131
132        if !keep_tracking {
133            break;
134        }
135    }
