18/// Routine used to compute a split factor, value used to regulate population
19/// of the problem.
20///
21/// If load balancing is enabled, the spawned particle will be spread
22/// throughout the threads to reach an even distribution. Else, the spawning/killing
23/// of particles will be uniform across threads, ignoring the local population.
24/// This is reflected in the split factor computation.
25pub fn compute_split_factor<T: CustomFloat>(
26    global_target_n_particles: usize,
27    global_n_particles: usize,
28    local_n_particles: usize,
29    num_threads: usize,
30    load_balance: bool,
31) -> T {
32    if load_balance {
33        // unit-specific modifications to reach uniform distribution
34        if local_n_particles != 0 {
35            let local_target_n_particles: usize = {
36                let tmp: T = <T as FromPrimitive>::from_usize(global_target_n_particles).unwrap()
37                    / FromPrimitive::from_usize(num_threads).unwrap();
38                tmp.ceil().to_usize().unwrap()
39            };
40            // return
41            <T as FromPrimitive>::from_usize(local_target_n_particles).unwrap()
42                / FromPrimitive::from_usize(local_n_particles).unwrap()
43        } else {
44            // return
45            one()
46        }
47    } else {
48        // uniform modifications across units without regards to distribution
49        // return
50        <T as FromPrimitive>::from_usize(global_target_n_particles).unwrap()
51            / FromPrimitive::from_usize(global_n_particles).unwrap()
52    }
53}

55/// Simulates the sources according to the problem's parameters.
56///
57/// This function spawns particle is source regions. Each time this function
58/// is called (once per cycle), 10% of the target number of particles are
59/// spawned. _Where_ they are spawned depends on both deterministic factors and
60/// randomness.
61///
62/// Each cell is given a weight, computed according to source rate of its
63/// material, volume and time step: The weight is directly proportional to
64/// the number of particle that should be spawned in the cell, during a time
65/// interval equal to the time step. Given this weight, and the weight of a
66/// fresh particle at sourcing, we can compute the number of particle the cell
67/// should spawn to get the overall desired number.
68///
69/// The amount of particle spawned is a consequence of the source particle weight
70/// computation as the value is adjusted according to the problem's total weight.
71pub fn source_now<T: CustomFloat>(
72    mcdata: &MonteCarloData<T>,
73    mcunit: &mut MonteCarloUnit<T>,
74    container: &mut ParticleContainer<T>,
75) {
76    let time_step = mcdata.params.simulation_params.dt;
77
78    let source_rate: Vec<T> = mcdata
79        .material_database
80        .mat
81        .iter()
82        .map(|mat| mcdata.params.material_params[&mat.name].source_rate)
83        .collect();
84    let mut seeds: Vec<u64> = Vec::new();
85
86    // on each domain
87    mcunit
88        .domain
89        .cell_state
90        .iter_mut()
91        .enumerate()
92        .for_each(|(cell_idx, cell)| {
93            // compute number of particles to be created in the cell
94            let cell_weight: T = cell.volume * source_rate[cell.material] * time_step;
95            let cell_n_particles: usize = (cell_weight / mcdata.source_particle_weight)
96                .floor()
97                .to_usize()
98                .unwrap();
99            seeds.clear();
100
101            for _ in 0..cell_n_particles {
102                // source_tally is fetched & incr atomically in original code
103                let rand_n_seed = (cell.source_tally + cell.id) as u64;
104                cell.source_tally += 1;
105                seeds.push(rand_n_seed);
106            }
107
108            mcunit.tallies.balance_cycle[TalliedEvent::Source] += seeds.len() as u64;
109
110            container
111                .processing_particles
112                .extend(seeds.iter_mut().map(|rand_n_seed| {
113                    // ~~~ init particle
114
115                    let mut particle: MCParticle<T> = MCParticle::default();
116
117                    particle.random_number_seed = spawn_rn_seed::<T>(rand_n_seed);
118                    particle.identifier = *rand_n_seed;
119                    particle.coordinate = generate_coordinate_3dg(
120                        &mut particle.random_number_seed,
121                        &mcunit.domain.mesh,
122                        cell_idx,
123                        cell.volume,
124                    );
125                    particle.domain = mcunit.domain.global_domain;
126                    particle.cell = cell_idx;
127                    particle.weight = mcdata.source_particle_weight;
128
129                    // ~~~ random sampling
130
131                    particle.sample_isotropic();
132                    // sample energy uniformly in [emin; emax] MeV
133                    let range = mcdata.params.simulation_params.e_max
134                        - mcdata.params.simulation_params.e_min;
135                    particle.kinetic_energy = rng_sample::<T>(&mut particle.random_number_seed)
136                        * range
137                        + mcdata.params.simulation_params.e_min;
138                    particle.sample_num_mfp();
139                    particle.time_to_census =
140                        time_step * rng_sample::<T>(&mut particle.random_number_seed);
141
142                    particle
143                }));
144        });
145}
