1//! Code for segment outcome computation, i.e. whichever event the particle
2//! undergoes next
3//!
4//! This module contains the function computing which event a given particle
5//! will undergo next. This is done by using the notion of _distance_ to an event
6//! and finding the minimum distance between all the possible events.\
7//! The particle's position and time to census is also updated, while
8//! event-specific modifications are left to event-specific functions. The result
9//! is returned using an enum ([`MCSegmentOutcome`])that takes value according to
10//! the event.
11

24};

26/// Enum representing the outcome of the current segment.
27#[derive(Debug, Clone, Copy, PartialEq)]
28pub enum MCSegmentOutcome {
29    /// Value for collision event.
30    Collision = 0,
31    /// Value for facet crossing event.
32    FacetCrossing = 1,
33    /// Value for census, i.e. no event.
34    Census = 2,
35}
36
37/// Structure used to handle all distance related data & comparison.
38pub struct DistanceHandler<T: CustomFloat> {
39    /// Distance to collision.
40    pub collision: T,
41    /// Distance to facet crossing.
42    pub facet_crossing: T,
43    /// Distance to census.
44    pub census: T,
45    /// Current minimum distance.
46    pub min_dist: T,
47    /// Current outcome.
48    pub outcome: MCSegmentOutcome,
49}
50
51impl<T: CustomFloat> DistanceHandler<T> {
52    /// Update the distance to a given outcome with the provided value.
53    /// Also check for a new minimum distance and update the structure
54    /// accordingly.
55    pub fn update(&mut self, which_outcome: MCSegmentOutcome, dist_outcome: T) {
56        match which_outcome {
57            MCSegmentOutcome::Collision => self.collision = dist_outcome,
58            MCSegmentOutcome::FacetCrossing => self.facet_crossing = dist_outcome,
59            MCSegmentOutcome::Census => self.census = dist_outcome,
60        }
61        if dist_outcome < self.min_dist {
62            self.min_dist = dist_outcome;
63            self.outcome = which_outcome;
64        }
65    }
66
67    /// Update the structure to force a collision
68    pub fn force_collision(&mut self) {
69        self.collision = T::tiny_float();
70        self.facet_crossing = T::huge_float();
71        self.census = T::huge_float();
72        self.min_dist = T::tiny_float();
73        self.outcome = MCSegmentOutcome::Collision;
74    }
75}
76
77impl<T: CustomFloat> Default for DistanceHandler<T> {
78    fn default() -> Self {
79        let huge_f: T = T::huge_float();
80        Self {
81            collision: huge_f,
82            facet_crossing: huge_f,
83            census: huge_f,
84            min_dist: huge_f,
85            outcome: MCSegmentOutcome::Collision,
86        }
87    }
88}

90/// Computes the outcome of the current segment for a given particle.
91///
92/// Three outcomes are possible for a given particle: census, facet crossing or
93/// collision. The distances are computed as follows:
94///
95/// - Census: The distance is simply equal to the speed of the particle multiplied
96///   by the time left until census.
97/// - Facet crossing: The distance is computed in a similar way by the function
98///   [`nearest_facet()`].
99/// - Collision: The distance is computed using probabilities.
100pub fn outcome<T: CustomFloat>(
101    mcdata: &MonteCarloData<T>,
102    mcunit: &MonteCarloUnit<T>,
103    particle: &mut MCParticle<T>,
104) -> MCSegmentOutcome {
105    //==========
106    // Prep work
107
108    // initialize distances and constants
109    let small_f: T = T::small_float();
110    let mut distance_handler = DistanceHandler::default();
111
112    let particle_speed = particle.get_speed();
113
114    let force_collision = particle.num_mean_free_paths < zero();
115    if force_collision {
116        particle.num_mean_free_paths = small_f;
117    }
118
119    // get cross-section
120    // lazily computed
121    // This ordering should make it so that we don't compute a XS multiple times?
122    let pcxs = mcunit.xs_cache[(particle.cell, particle.energy_group)].load(Ordering::Acquire);
123    let macroscopic_total_xsection = if pcxs > zero() {
124        // use precomputed value
125        pcxs
126    } else {
127        // compute & cache value
128        let mat_gid: usize = mcunit.domain.cell_state[particle.cell].material;
129        let cell_nb_density: T = mcunit.domain.cell_state[particle.cell].cell_number_density;
130        let tmp = weighted_macroscopic_cross_section(
131            mcdata,
132            mat_gid,
133            cell_nb_density,
134            particle.energy_group,
135        );
136        mcunit.xs_cache[(particle.cell, particle.energy_group)].store(tmp, Ordering::Release);
137        tmp
138    };
139
140    // prepare particle
141    particle.total_cross_section = macroscopic_total_xsection;
142    particle.mean_free_path = if macroscopic_total_xsection == zero() {
143        T::huge_float()
144    } else {
145        one::<T>() / macroscopic_total_xsection
146    };
147
148    // if zero
149    if particle.num_mean_free_paths == zero() {
150        particle.sample_num_mfp();
151    }
152
153    //===========================
154    // Compute distances to event
155
156    // collision
157    distance_handler.update(
158        MCSegmentOutcome::Collision,
159        particle.num_mean_free_paths * particle.mean_free_path,
160    );
161    // census
162    distance_handler.update(
163        MCSegmentOutcome::Census,
164        particle_speed * particle.time_to_census,
165    );
166    // nearest facet
167    let nearest_facet: MCNearestFacet<T> = nearest_facet(particle, &mcunit.domain.mesh);
168    particle.normal_dot = nearest_facet.dot_product;
169    distance_handler.update(
170        MCSegmentOutcome::FacetCrossing,
171        nearest_facet.distance_to_facet,
172    );
173
174    // force a collision if needed
175    if force_collision {
176        distance_handler.force_collision();
177    }
178
179    //============================================
180    // Pick the outcome & update particle, tallies
181
182    let segment_outcome = distance_handler.outcome;
183    assert!(distance_handler.min_dist >= zero());
184
185    // general update
186    particle.segment_path_length = distance_handler.min_dist;
187    particle.num_mean_free_paths -= particle.segment_path_length / particle.mean_free_path;
188
189    // outcome-specific updates
190    match segment_outcome {
191        MCSegmentOutcome::Collision => {
192            particle.num_mean_free_paths = zero();
193            particle.last_event = MCTallyEvent::Collision;
194        }
195        MCSegmentOutcome::FacetCrossing => {
196            particle.facet = nearest_facet.facet;
197            particle.last_event = MCTallyEvent::FacetCrossingTransitExit;
198        }
199        MCSegmentOutcome::Census => {
200            particle.time_to_census = zero::<T>().min(particle.time_to_census);
201            particle.last_event = MCTallyEvent::Census;
202        }
203    }
204
205    // skip tallies & early return if the path length is 0
206    if particle.segment_path_length == zero() {
207        return segment_outcome;
208    }
209
210    // move particle to the end of the segment
211    particle.move_particle_along_segment();
212
213    // decrement time to census & increment age
214    let segment_path_time = particle.segment_path_length / particle_speed;
215    particle.time_to_census -= segment_path_time;
216    particle.age += segment_path_time;
217    particle.time_to_census = particle.time_to_census.max(zero());
218
219    segment_outcome
220}
