1//! Code for cross-section computation a.k.a. probability density computation
2//!
3//! This module contains function used to compute cross-section data from known
4//! physical quantities of the material and reacting isotope.
5
6use num::{zero, FromPrimitive};
7
8use crate::{constants::CustomFloat, montecarlo::MonteCarloData};
9
10/// Computes the total number-density-weighted macroscopic
11/// cross-section in the cell.
12///
13/// Note that this function is isotope-specific; However the proxy-app
14/// only accounts for simulation of a single isotope type.
15fn macroscopic_total_cross_section<T: CustomFloat>(
16    mcdata: &MonteCarloData<T>,
17    mat_gid: usize,
18    isotope_idx: usize,
19    cell_nb_density: T,
20    energy_group: usize,
21) -> T {
22    let atom_fraction: T = mcdata.material_database.mat[mat_gid].iso[isotope_idx].atom_fraction;
23
24    if (atom_fraction == zero()) | (cell_nb_density == zero()) {
25        // one of the two is 0
26        let res: T = FromPrimitive::from_f64(1e-20).unwrap();
27        return res;
28    }
29
30    let isotope_gid = mcdata.material_database.mat[mat_gid].iso[isotope_idx].gid;
31    let micro_cross_section: T = mcdata
32        .nuclear_data
33        .get_total_cross_section(isotope_gid, energy_group);
34
35    atom_fraction * cell_nb_density * micro_cross_section
36}

38/// Computes the number-density-weighted macroscopic cross-section
39/// of a collection of isotopes in the cell.
40///
41/// Note that there is not really any weighting, this comes from the original
42/// choice of Quicksilver to leave material weighting out of the proxy-app.
43pub fn weighted_macroscopic_cross_section<T: CustomFloat>(
44    mcdata: &MonteCarloData<T>,
45    mat_gid: usize,
46    cell_nb_density: T,
47    energy_group: usize,
48) -> T {
49    let n_isotopes: usize = mcdata.material_database.mat[mat_gid].iso.len();
50
51    (0..n_isotopes)
52        .map(|isotope_idx| {
53            macroscopic_total_cross_section(
54                mcdata,
55                mat_gid,
56                isotope_idx,
57                cell_nb_density,
58                energy_group,
59            )
60        })
61        .sum()
62}
