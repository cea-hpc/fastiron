8//!

48/// Associated reference value used for compute approximation.
49///
50/// This is the only trait that should be manually implemented for custom types. The `CustomFloat` trait is then
51/// automatically implemented via a blanket implementation.
52pub trait CustomReferenceFloat: Float + FromPrimitive {
53    /// Threshold upper-value for decimal number.
54    const HUGE_FLOAT: Self;
55    /// Threshold low-ish-value for decimal number.
56    const SMALL_FLOAT: Self;
57    /// Threshold lower-value for decimal number.
58    const TINY_FLOAT: Self;
59    /// Pi value.
60    const PI: Self;
61}
62
63/// Custom trait for floating point number
64pub trait CustomFloat:
65    Float
66    + CustomReferenceFloat
67    + bytemuck::Pod
68    + Default
69    // conversions
70    + FromPrimitive
71    + FromStr
72    // operations
73    + AddAssign
74    + SubAssign
75    + MulAssign
76    + DivAssign
77    + Sum
78    // display
79    + Debug
80    + Display
81    + LowerExp
82    // parallel-safe
83    + Send
84    + Sync
85{
86    /// Threshold upper-value for decimal number.
87    fn huge_float<T: CustomFloat>() -> T {
88        T::HUGE_FLOAT
89    }
90    /// Threshold low-ish-value for decimal number.
91    fn small_float<T: CustomFloat>() -> T {
92        T::SMALL_FLOAT
93    }
94    /// Threshold lower-value for decimal number.
95    fn tiny_float<T: CustomFloat>() -> T {
96        T::TINY_FLOAT
97    }
98
99    fn neutron_mass_energy<T: CustomFloat>() -> T {
100        T::from(9.39565e+2).unwrap()
101    }
102
103    fn pi<T: CustomFloat>() -> T {
104        T::from(T::PI).unwrap()
105    }
106
107    fn light_speed<T: CustomFloat>() -> T {
108        T::from(2.99792e+10).unwrap()
109    }
110}

112impl CustomReferenceFloat for f32 {
113    /// Threshold value when using [f32]. May need adjustment.
114    const HUGE_FLOAT: Self = 10e35_f32;
115
116    /// Threshold value when using [f32]. May need adjustment.
117    const SMALL_FLOAT: Self = 1e-10_f32;
118
119    /// Threshold value when using [f32]. May need adjustment.
120    const TINY_FLOAT: Self = 1e-13_f32;
121
122    /// Pi value when using [f32].
123    const PI: Self = std::f32::consts::PI;
124}
