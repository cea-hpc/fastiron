23/// Custom enum used to model a particle's species.

25pub enum Species {

33/// Structure used to hold all data of a particle.
34///
35/// This is mostly used for computations during the tracking section.
36#[derive(Debug, Default, PartialOrd, PartialEq, Clone)]
37pub struct MCParticle<T: CustomFloat> {
38    /// Current position.
39    pub coordinate: MCVector<T>,
40    /// Direction of the particle as a normalized `(x, y, z)` vector.
41    pub direction: MCVector<T>,
42    /// Kinetic energy.
43    pub kinetic_energy: T,
44    /// Current energy group the particle belong to.
45    pub energy_group: usize,
46    /// Weight.
47    pub weight: T,
48    /// Species of the particle.
49    pub species: Species,
50
51    /// Time remaining before this particle hit census.
52    pub time_to_census: T,
53    /// Age.
54    pub age: T,
55
56    /// Cache-ing the current total cross-section.
57    pub total_cross_section: T,
58    /// Distance to a collision.
59    pub mean_free_path: T,
60    /// Distance this particle travels in a segment.
61    pub segment_path_length: T,
62    /// Number of mean free paths to a collision.
63    pub num_mean_free_paths: T,
64    /// Number of segments the particle travelled.
65    pub num_segments: T,
66
67    /// Random number seed for the rng for this particle.
68    pub random_number_seed: u64,
69    /// Unique ID used to identify and track individual particles.
70    pub identifier: u64,
71    /// Last event this particle underwent.
72    pub last_event: MCTallyEvent,
73
74    /// Current domain in the spatial grid.
75    pub domain: usize,
76    /// Current cell in the current domain.
77    pub cell: usize,
78    /// Nearest facet.
79    pub facet: usize,
80    /// Normal dot product value kept when crossing a facet.
81    pub normal_dot: T,
82}

84impl<T: CustomFloat> MCParticle<T> {
85    /// Update the particle's field to model its movement along its
86    /// direction, for the segment length.
87    pub fn move_particle_along_segment(&mut self) {
88        self.coordinate += self.direction * self.segment_path_length;
89    }

91    /// Update the particle's trajectory with new energy & angle.
92    pub fn update_trajectory(&mut self, energy: T, angle: T) {
93        // constants
94        let pi: T = T::pi();
95        let one: T = one();
96        let two: T = FromPrimitive::from_f64(2.0).unwrap();
97
98        // value for update
99        let cos_theta: T = angle;
100        let sin_theta: T = (one - cos_theta * cos_theta).sqrt();
101        let rdm_number: T = rng_sample(&mut self.random_number_seed);
102        let phi = two * pi * rdm_number;
103        let sin_phi: T = phi.sin();
104        let cos_phi: T = phi.cos();
105
106        // update
107        self.kinetic_energy = energy;
108        self.rotate_direction(sin_theta, cos_theta, sin_phi, cos_phi);
109        self.sample_num_mfp();
110    }
111
112    /// Computes the particle speed from its energy. Note that this computation
113    /// should be species-specific.
114    pub fn get_speed(&self) -> T {
115        let rest_mass_energy: T = T::neutron_mass_energy();
116        let speed_of_light: T = T::light_speed();
117        let two: T = FromPrimitive::from_f64(2.0).unwrap();
118        speed_of_light
119            * (self.kinetic_energy * (self.kinetic_energy + two * (rest_mass_energy))
120                / ((self.kinetic_energy + rest_mass_energy)
121                    * (self.kinetic_energy + rest_mass_energy)))
122                .sqrt()
123    }

125    /// Return the starting cross-section for reaction sampling.
126    pub fn get_current_xs(&mut self) -> T {
127        self.total_cross_section * rng_sample::<T>(&mut self.random_number_seed)
128    }

130    /// Uses a PRNG to sample new energy & angle after a reaction.
131    ///
132    /// Since reaction type is specified when the method is called, we assume
133    /// that the result will be treated correctly by the calling code.
134    pub fn sample_collision(
135        &mut self,
136        reaction: &NuclearDataReaction<T>,
137        material_mass: T,
138        extra: &mut ParticleCollection<T>,
139    ) -> usize {
140        let one: T = FromPrimitive::from_f64(1.0).unwrap();
141        let two: T = FromPrimitive::from_f64(2.0).unwrap();
142        // Need to replace with tinyvec types
143        match reaction.reaction_type {
144            ReactionType::Scatter => {
145                let energy = self.kinetic_energy
146                    * (one - rng_sample::<T>(&mut self.random_number_seed) * (one / material_mass));
147                let angle = rng_sample::<T>(&mut self.random_number_seed) * two - one;
148                self.update_trajectory(energy, angle);
149                1
150            }
151            ReactionType::Absorption => 0,
152            ReactionType::Fission => {
153                let num_particle_out: usize = (reaction.nu_bar
154                    + rng_sample(&mut self.random_number_seed))
155                .to_usize()
156                .unwrap();
157                let twenty: T = FromPrimitive::from_f64(20.0).unwrap(); // should be Emax ?
158                match num_particle_out {
159                    0 => (),
160                    1 => {
161                        let rand_f = (rng_sample::<T>(&mut self.random_number_seed) + one) / two;
162                        let energy = twenty * rand_f * rand_f;
163                        let angle = rng_sample::<T>(&mut self.random_number_seed) * two - one;
164                        self.update_trajectory(energy, angle);
165                    }
166                    _ => {
167                        assert!(num_particle_out < 5); // this is guaranteed by the way we sample and the nu bar value
168
169                        // for the original particle
170                        let rand_f = (rng_sample::<T>(&mut self.random_number_seed) + one) / two;
171                        let energy = twenty * rand_f * rand_f;
172                        let angle = rng_sample::<T>(&mut self.random_number_seed) * two - one;
173
174                        let mut out = ArrayVec::<[(T, T); 5]>::default();
175                        out.extend((1..num_particle_out).map(|_| {
176                            let rand_f =
177                                (rng_sample::<T>(&mut self.random_number_seed) + one) / two;
178                            let energy_out = twenty * rand_f * rand_f;
179                            let angle_out =
180                                rng_sample::<T>(&mut self.random_number_seed) * two - one;
181                            (energy_out, angle_out)
182                        }));
183
184                        let mut seeds = ArrayVec::<[u64; 5]>::default();
185                        seeds.extend(
186                            (1..num_particle_out)
187                                .map(|_| spawn_rn_seed::<T>(&mut self.random_number_seed)),
188                        );
189
190                        let sec_particles = zip(seeds, out).map(|(seed, (energy, angle))| {
191                            let mut sec_particle = self.clone();
192                            sec_particle.random_number_seed = seed;
193                            sec_particle.identifier = seed;
194                            sec_particle.update_trajectory(energy, angle);
195                            sec_particle
196                        });
197                        extra.extend(sec_particles);
198
199                        self.update_trajectory(energy, angle);
200                    }
201                }
202                num_particle_out
203            }
204        }
205    }

207    /// Sample the number of mean free paths to a collision.
208    pub fn sample_num_mfp(&mut self) {
209        self.num_mean_free_paths = -one::<T>() * rng_sample::<T>(&mut self.random_number_seed).ln();
210    }

212    /// Sample a random direction for the particle to face.
213    pub fn sample_isotropic(&mut self) {
214        let one: T = one();
215        let two: T = FromPrimitive::from_f64(2.0).unwrap();
216        let pi: T = T::pi();
217
218        // sample gamma
219        self.direction.z = one - two * rng_sample(&mut self.random_number_seed);
220        let sine_gamma = (one - self.direction.z * self.direction.z).sqrt();
221
222        // sample phi and set the other angles using it
223        let phi = pi * (two * rng_sample(&mut self.random_number_seed) - one);
224
225        self.direction.x = sine_gamma * phi.cos();
226        self.direction.y = sine_gamma * phi.sin();
227    }

229    /// Rotates a 3D vector that is defined by the angles Theta and Phi
230    /// in a local coordinate frame about a polar angle and azimuthal angle
231    /// described by the direction cosine. Hence, caller passes in
232    /// `sin_Theta` and `cos_Theta` referenced from the local z-axis and `sin_Phi`
233    /// and `cos_Phi` referenced from the local x-axis to describe the vector V
234    /// to be rotated. The direction cosine describes global theta and phi
235    /// angles that the vector V is to be rotated about.
236    /// `cos_theta_zero`/`sin_theta_zero` and `cos_phi_zero`/`sin_phi_zero`
237    /// model the initial position while the arguments of the method characterize
238    /// the rotation. See [this][1] for explanation on the formula.
239    ///
240    /// [1]: https://en.wikipedia.org/wiki/Spherical_coordinate_system#Integration_and_differentiation_in_spherical_coordinates
241    pub fn rotate_direction(&mut self, sine_theta: T, cosine_theta: T, sine_phi: T, cosine_phi: T) {
242        let one: T = one();
243        let threshold: T = FromPrimitive::from_f64(1e-6).unwrap(); // order of TINY_FLOAT.sqrt()
244
245        let cos_theta_zero = self.direction.z;
246        let sin_theta_zero = (one - cos_theta_zero * cos_theta_zero).sqrt();
247
248        let (cos_phi_zero, sin_phi_zero): (T, T) = if sin_theta_zero < threshold {
249            (one, zero())
250        } else {
251            (
252                self.direction.x / sin_theta_zero,
253                self.direction.y / sin_theta_zero,
254            )
255        };
256
257        // compute the rotation
258        self.direction.x = cos_theta_zero * cos_phi_zero * (sine_theta * cosine_phi)
259            - sin_phi_zero * (sine_theta * sine_phi)
260            + sin_theta_zero * cos_phi_zero * cosine_theta;
261
262        self.direction.y = cos_theta_zero * sin_phi_zero * (sine_theta * cosine_phi)
263            + cos_phi_zero * (sine_theta * sine_phi)
264            + sin_theta_zero * sin_phi_zero * cosine_theta;
265
266        self.direction.z =
267            -sin_theta_zero * (sine_theta * cosine_phi) + zero() + cos_theta_zero * cosine_theta;
268    }

270    /// Returns an iterator over particles created from a split of the original one (caller).
271    /// This is used in the population control algorithm.
272    pub fn under_populated_split(
273        &mut self,
274        split_rr_factor: T,
275    ) -> impl Iterator<Item = MCParticle<T>> + '_ {
276        let mut split_factor = split_rr_factor.floor();
277        if rng_sample::<T>(&mut self.random_number_seed) > split_rr_factor - split_factor {
278            split_factor -= one();
279        }
280        self.weight /= split_rr_factor;
281
282        let n_split: usize = split_factor.to_usize().unwrap();
283        // return an iterating object, so we can use .flat_map() method
284        (0..n_split).map(|_| {
285            let mut split_pp = self.clone();
286            split_pp.random_number_seed = spawn_rn_seed::<T>(&mut self.random_number_seed);
287            split_pp.identifier = split_pp.random_number_seed;
288
289            split_pp
290        })
291    }

293    /// Play russian-roulette with the particle, returning true if the particle survives,
294    /// false otherwise. This function is meant to be used along the [`Vec::retain_mut()`]
295    /// method.
296    pub fn over_populated_rr(&mut self, split_rr_factor: T) -> bool {
297        if rng_sample::<T>(&mut self.random_number_seed) > split_rr_factor {
298            // particle dies
299            false
300        } else {
301            // particle survives with increased weight
302            self.weight /= split_rr_factor;
303            true
304        }
305    }
306
307    /// Play russian-roulette with particle of low statistical weight, returning true if
308    /// the particle survives, false otherwise. This function is meant to be used along
309    /// the [`Vec::retain_mut()`] method.
310    pub fn low_weight_rr(&mut self, relative_weight_cutoff: T, source_particle_weight: T) -> bool {
311        let weight_cutoff = relative_weight_cutoff * source_particle_weight;
312        if self.weight <= weight_cutoff {
313            if rng_sample::<T>(&mut self.random_number_seed) <= relative_weight_cutoff {
314                // particle survives with increased weight
315                self.weight /= relative_weight_cutoff;
316                true
317            } else {
318                // particle dies
319                false
320            }
321        } else {
322            // particle survives
323            true
324        }
325    }

355        let mut pp: MCParticle<f64> = MCParticle::default();
356        // init data
357        let e: f64 = 1.0;

365        let energy = rng_sample(&mut seed);
366        let angle = rng_sample(&mut seed);
