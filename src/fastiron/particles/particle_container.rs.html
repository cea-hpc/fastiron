19#[derive(Debug, Clone)]
20/// Structure used as a container for all particles.
21///
22/// The [Clone] implementation should not be used except at the beginning of the program.
23pub struct ParticleContainer<T: CustomFloat> {
24    /// Container for particles that have yet to be processed.
25    pub processing_particles: ParticleCollection<T>,
26    /// Container for already processed particles.
27    pub processed_particles: ParticleCollection<T>,
28    /// Container for extra particles. This is used for fission-induced
29    /// particles and incoming off-processor particles.
30    pub extra_particles: ParticleCollection<T>,
31}
32
33impl<T: CustomFloat> ParticleContainer<T> {
34    /// Constructor. The appropriate capacity is computed beforehand.
35    pub fn new(regular_capacity: usize, extra_capacity: usize) -> Self {
36        Self {
37            processing_particles: ParticleCollection::with_capacity(regular_capacity),
38            processed_particles: ParticleCollection::with_capacity(regular_capacity),
39            extra_particles: ParticleCollection::with_capacity(extra_capacity),
40        }
41    }
42
43    /// Swap the processing and processed particle lists. This function is used in-between
44    /// iterations.
45    pub fn swap_processing_processed(&mut self) {
46        self.processing_particles
47            .append(&mut self.processed_particles);
48    }
49
50    /// Randomly delete particles to reach the desired number of particles for
51    /// simulation. Low weight particles are, then, randomly deleted.
52    pub fn regulate_population(
53        &mut self,
54        split_rr_factor: T,
55        relative_weight_cutoff: T,
56        source_particle_weight: T,
57        balance: &mut Balance,
58    ) {
59        let old_len = self.processing_particles.len();
60        self.processing_particles.retain_mut(|pp| {
61            let survive_once = pp.over_populated_rr(split_rr_factor);
62            let survive_twice = pp.low_weight_rr(relative_weight_cutoff, source_particle_weight);
63            survive_once & survive_twice
64        });
65        balance[TalliedEvent::OverRr] = (old_len - self.processing_particles.len()) as u64;
66    }
67
68    /// Split particles to reach the desired number of particles for
69    /// simulation. Low weight particles are, then, randomly deleted.
70    pub fn split_population(
71        &mut self,
72        split_rr_factor: T,
73        relative_weight_cutoff: T,
74        source_particle_weight: T,
75        balance: &mut Balance,
76    ) {
77        let mut old_len = self.processing_particles.len();
78        (&mut self.processing_particles).into_iter().for_each(|pp| {
79            self.extra_particles
80                .extend(pp.under_populated_split(split_rr_factor));
81        });
82        self.clean_extra_vaults();
83        balance[TalliedEvent::Split] = (self.processing_particles.len() - old_len) as u64;
84        old_len = self.processing_particles.len();
85        self.processing_particles
86            .retain_mut(|pp| pp.low_weight_rr(relative_weight_cutoff, source_particle_weight));
87        balance[TalliedEvent::WeightRr] = (old_len - self.processing_particles.len()) as u64;
88    }
89
90    /// Track particles and transfer them to the processed storage when done.
91    pub fn process_particles(
92        &mut self,
93        mcdata: &MonteCarloData<T>,
94        mcunit: &mut MonteCarloUnit<T>,
95    ) {
96        let exinf = &mcdata.exec_info;
97        match exinf.exec_policy {
98            // Process unit sequentially
99            ExecPolicy::Sequential | ExecPolicy::Distributed => {
100                let mut tmp = Balance::default();
101                (&mut self.processing_particles)
102                    .into_iter()
103                    .for_each(|particle| {
104                        cycle_tracking_guts(
105                            mcdata,
106                            mcunit,
107                            particle,
108                            &mut tmp,
109                            &mut self.extra_particles,
110                        )
111                    });
112                mcunit.tallies.balance_cycle.add_to_self(&tmp);
113            }
114            // Process unit in parallel
115            ExecPolicy::Rayon | ExecPolicy::Hybrid => {
116                let extra = Arc::new(Mutex::new(&mut self.extra_particles));
117                // choose chunk size to get one chunk per thread
118                let chunk_size: usize = match exinf.chunk_size {
119                    0 => (self.processing_particles.len() / exinf.n_rayon_threads) + 1,
120                    _ => exinf.chunk_size,
121                };
122
123                let res: Balance = self
124                    .processing_particles
125                    .par_iter_mut()
126                    .chunks(chunk_size)
127                    .map(|mut particles| {
128                        // Strategy used to reduce resource (memory) contention
129                        // 1. Give each chunks (==thread with our chunk_size value) its own balance
130                        //    This removes the need for atomics type. The tradeoff: folding results
131                        //    of the iterators
132                        // 2. Use a local extra collection that is later used to extend the global
133                        //    container. This reduces the total number of lock (and probably
134                        //    lock time)
135                        let mut local_balance: Balance = Balance::default();
136                        // chunk_size * 5 is enough capacity to handle all particles undergoing
137                        // fission & splitting into the max possible nb of particles.
138                        let mut local_extra: ParticleCollection<T> =
139                            ParticleCollection::with_capacity(chunk_size * 5);
140                        particles.iter_mut().for_each(|particle| {
141                            cycle_tracking_guts(
142                                mcdata,
143                                mcunit,
144                                particle,
145                                &mut local_balance,
146                                &mut local_extra,
147                            )
148                        });
149                        extra.lock().unwrap().append(&mut local_extra);
150                        local_balance
151                    })
152                    .fold_with(Balance::default(), |a, b| a + b)
153                    .sum::<Balance>();
154                // It should be safe to simply add this to the one in mcunit
155                assert_eq!(res[TalliedEvent::Start], 0);
156                assert_eq!(res[TalliedEvent::End], 0);
157                assert_eq!(res[TalliedEvent::Source], 0);
158                mcunit.tallies.balance_cycle.add_to_self(&res);
159            }
160        }
161        self.processing_particles
162            .retain(|particle| particle.species != Species::Unknown);
163        self.processed_particles
164            .append(&mut self.processing_particles);
165    }
166
167    /// Sort the processing particles according to where they belong in the mesh.
168    pub fn sort_processing(&mut self) {
169        self.processing_particles
170            .sort_by(|a, b| match a.cell.cmp(&b.cell) {
171                std::cmp::Ordering::Less => std::cmp::Ordering::Less,
172                std::cmp::Ordering::Equal => a.energy_group.cmp(&b.energy_group),
173                std::cmp::Ordering::Greater => std::cmp::Ordering::Greater,
174            });
175    }
176
177    /// Adds back to the processing storage the extra particles.
178    pub fn clean_extra_vaults(&mut self) {
179        self.processing_particles.append(&mut self.extra_particles);
180    }
181
182    /// Checks if there are no more particles to process, i.e:
183    /// - extra storage is empty
184    /// - processing storage is empty
185    /// - send queue is empty
186    pub fn is_done_processing(&self) -> bool {
187        self.extra_particles.is_empty() & self.processing_particles.is_empty()
188    }
189}
