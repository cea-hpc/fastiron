16type MapType = FxHashMap<usize, CellInfo>;
17type SetType = FxHashSet<usize>;
18
19/// Structure used to hold cell information.
20#[derive(Debug, Clone, Copy, Default)]
21pub struct CellInfo {
22    /// Domain global identifier
23    pub domain_gid: Option<usize>,
24    /// Foreman identifier
25    pub foreman: Option<usize>,
26    /// Domain index
27    pub domain_index: Option<usize>,
28    /// Cell index
29    pub cell_index: Option<usize>,
30}
31
32/// Structure used to represent the mesh partition of the space.
33///
34/// Holds all the different cells' information.
35#[derive(Debug, Clone)]
36pub struct MeshPartition {
37    /// Domain global identifier.
38    pub domain_gid: usize,
39    /// Foreman identifier.
40    pub foreman: usize,
41    /// Map linking cell global identifier to their [CellInfo] structure
42    pub cell_info_map: MapType,
43    /// List of neighboring domain identifiers.
44    pub nbr_domains: Vec<usize>,
45}
46
47impl MeshPartition {
48    /// Constructor. The structure is **not** ready to be used directly.
49    pub fn new(domain_gid: usize, foreman: usize) -> Self {
50        Self {
51            domain_gid,
52            foreman,
53            cell_info_map: Default::default(),
54            nbr_domains: Default::default(),
55        }
56    }
57    /// Builds the mesh partition. This method needs to be called after the
58    /// constructor.
59    pub fn build_mesh_partition<T: CustomFloat>(
60        &mut self,
61        grid: &GlobalFccGrid<T>,
62        centers: &[MCVector<T>],
63    ) -> Vec<(usize, usize)> {
64        self.assign_cells_to_domain(centers, grid);
65
66        self.build_cell_idx_map(grid)
67    }
68
69    /// Internal function that associates cells to domains
70    fn assign_cells_to_domain<T: CustomFloat>(
71        &mut self,
72        domain_center: &[MCVector<T>],
73        grid: &GlobalFccGrid<T>,
74    ) {
75        let mut assigner = GridAssignmentObject::new(domain_center);
76        let mut flood_queue: VecDeque<usize> = VecDeque::new();
77        let mut wet_cells: SetType = SetType::default();
78
79        let root = grid.which_cell(&domain_center[self.domain_gid]);
80
81        flood_queue.push_back(root);
82        wet_cells.insert(root);
83        Self::add_nbrs_to_flood(root, grid, &mut flood_queue, &mut wet_cells);
84
85        while !flood_queue.is_empty() {
86            let cell_idx = flood_queue.pop_front().unwrap();
87            let rr = grid.cell_center(cell_idx);
88            let domain = assigner.nearest_center(rr);
89
90            // insert only if the key is absent; in c++ there's no overwriting of keys
91            self.cell_info_map.entry(cell_idx).or_insert(CellInfo {
92                domain_gid: Some(domain),
93                ..Default::default()
94            });
95
96            if domain == self.domain_gid {
97                // if current cell is in domain, check neighbor
98                Self::add_nbrs_to_flood(cell_idx, grid, &mut flood_queue, &mut wet_cells);
99            } else if !self.nbr_domains.contains(&domain) {
100                // else, keep track of neighbor domains
101                self.nbr_domains.push(domain);
102            }
103        }
104    }
105
106    fn build_cell_idx_map<T: CustomFloat>(
107        &mut self,
108        grid: &GlobalFccGrid<T>,
109    ) -> Vec<(usize, usize)> {
110        let mut remote_cells: Vec<(usize, usize)> = Default::default();
111
112        let mut n_local_cells: usize = 0;
113
114        // TWICE THE MEMORY
115        let read_map = self.cell_info_map.clone();
116
117        for (cell_gid, cell_info) in &mut self.cell_info_map {
118            let domain_gid: usize = cell_info.domain_gid.unwrap();
119            // Always true when not distributed
120            if domain_gid == self.domain_gid {
121                // local cell
122                cell_info.cell_index = Some(n_local_cells);
123                n_local_cells += 1;
124                cell_info.domain_index = Some(self.domain_gid);
125                cell_info.foreman = Some(self.foreman);
126            } else {
127                // THIS IS THE PART THAT USES TWICE THE MEMORY
128                // WILL REMOVE WHEN REMOVING ALL TRACES OF DISTRIBUTED IMPLEMENTATION
129                println!("test");
130                let face_nbr = grid.get_face_nbr_gids(*cell_gid);
131                for j_cell_gid in face_nbr {
132                    if let Some(c_info) = read_map.get(&j_cell_gid) {
133                        if c_info.domain_gid != Some(self.domain_gid) {
134                            continue;
135                        }
136                        // replace the update to sendSet
137                        if !remote_cells.contains(&(domain_gid, j_cell_gid)) {
138                            remote_cells.push((domain_gid, j_cell_gid));
139                        }
140                    }
141                }
142            }
143        }
144
145        // processing of the return value replaces comm.exchange
146        // remote cells are the cells of the CURRENT partition that
147        // are neighbors to the neighbor partition
148        // hence they are supposed to be inserted in the corresponding
149        // neighbor partition as they are (they contain correct info).
150        remote_cells
151    }
152
153    fn add_nbrs_to_flood<T: CustomFloat>(
154        cell_idx: usize,
155        grid: &GlobalFccGrid<T>,
156        flood_queue: &mut VecDeque<usize>,
157        wet_cells: &mut SetType,
158    ) {
159        let tt: Tuple3 = grid.cell_idx_to_tuple(cell_idx);
160        const NBR_COORDS: [(i32, i32, i32); 26] = [
161            // (-1, x, y)
162            (-1, -1, -1),
163            (-1, -1, 0),
164            (-1, -1, 1),
165            (-1, 0, -1),
166            (-1, 0, 0),
167            (-1, 0, 1),
168            (-1, 1, -1),
169            (-1, 1, 0),
170            (-1, 1, 1),
171            // (0, x, y) except (0, 0, 0)
172            (0, -1, -1),
173            (0, -1, 0),
174            (0, -1, 1),
175            (0, 0, -1),
176            (0, 0, 1),
177            (0, 1, -1),
178            (0, 1, 0),
179            (0, 1, 1),
180            // (1, x, y)
181            (1, -1, -1),
182            (1, -1, 0),
183            (1, -1, 1),
184            (1, 0, -1),
185            (1, 0, 0),
186            (1, 0, 1),
187            (1, 1, -1),
188            (1, 1, 0),
189            (1, 1, 1),
190        ];
191
192        NBR_COORDS
193            .iter()
194            .map(|offset| {
195                let snaped = grid.snap_turtle((
196                    tt.0 as i32 + offset.0,
197                    tt.1 as i32 + offset.1,
198                    tt.2 as i32 + offset.2,
199                ));
200                grid.cell_tuple_to_idx(&snaped)
201            })
202            .for_each(|nbr_idx| {
203                if wet_cells.insert(nbr_idx) {
204                    // the method return true if the object was inserted,
205                    // i.e. if it was absent before
206                    flood_queue.push_back(nbr_idx);
207                }
208            });
209    }
210}

219    use crate::{data::mc_vector::MCVector, geometry::global_fcc_grid::GlobalFccGrid};

223    #[test]
224    fn partition_building() {
225        // simple grid 2*2*2 grid, each cell dim is 1
226        let grid = GlobalFccGrid::new(2, 2, 2, 2.0, 2.0, 2.0);
227        // 2 symmetrical centers
228        let c1 = MCVector {
229            x: 0.0,
230            y: 0.0,
231            z: 0.0,
232        };
233        let c2 = MCVector {
234            x: 2.0,
235            y: 2.0,
236            z: 2.0,
237        };
238        let centers = vec![c1, c2];
239        let domain_gids: Vec<usize> = vec![0, 1];
240        let mut partition: Vec<MeshPartition> = Vec::with_capacity(centers.len());
241        domain_gids.iter().for_each(|ii| {
242            partition.push(MeshPartition::new(*ii, 0));
243        });
244
245        (0..partition.len()).for_each(|part_idx| {
246            let remote_cells = partition[part_idx].build_mesh_partition(&grid, &centers);
247            // only 2 domains, we can manually process those; gids and indexes are coherent
248            println!("{} remote cells", remote_cells.len());
249            // remote cells are a special case where we want to overwrite the target map's entry
250            remote_cells
251                .iter()
252                .for_each(|(remote_domain_gid, cell_gid)| {
253                    let cell_to_insert = partition[part_idx].cell_info_map[cell_gid];
254                    partition[*remote_domain_gid]
255                        .cell_info_map
256                        .insert(*cell_gid, cell_to_insert);
257
258                    println!("remote domain: {remote_domain_gid}");
259                    println!("cell (gid {cell_gid}): {cell_to_insert:#?}");
260                });
261
262            println!("{:#?}", partition[part_idx]);
263            println!();
264        });
265
266        // NOTE: only the belonging and neighboring cells are initialized
267        // NOTE: is there a way to test this or remove non-neighboring cells? is it worth it?
268        // for this simple case, non-neighboring cell are gid 0 in domain 1 and
269        // gid 7 in domain 0
270        partition.iter().for_each(|part| {
271            part.cell_info_map.iter().for_each(|(cell_gid, cell_info)| {
272                if ((*cell_gid == 0) & (part.domain_gid == 1))
273                    || ((*cell_gid == 7) & (part.domain_gid == 0))
274                {
275                    assert!(cell_info.cell_index.is_none());
276                    assert!((cell_info.domain_index.is_none()));
277                    assert!((cell_info.foreman.is_none()));
278                    return;
279                }
280                assert!((cell_info.domain_gid.is_some()));
281                assert!((cell_info.cell_index.is_some()));
282                assert!((cell_info.domain_index.is_some()));
283                assert!((cell_info.foreman.is_some()));
284            });
285        });
286    }
