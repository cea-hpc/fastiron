1//! Code used to build and navigate the basis of the mesh
2use num::{zero, FromPrimitive};
3
4use super::{CORNER_OFFSET, FACE_OFFSET, N_FACES, N_POINTS_INTERSEC};
5use crate::{
6    constants::{CustomFloat, Tuple3, Tuple4},
7    data::mc_vector::MCVector,
8};
9
10/// Structure representing the spatial grid of the problem.
11#[derive(Debug)]
12pub struct GlobalFccGrid<T: CustomFloat> {
13    /// Number of cells along the x-axis.
14    pub nx: usize,
15    /// Number of cells along the y-axis.
16    pub ny: usize,
17    /// Number of cells along the z-axis.
18    pub nz: usize,
19
20    /// Size of the problem along the x-axis (cm).
21    pub lx: T,
22    /// Size of the problem along the y-axis (cm).
23    pub ly: T,
24    /// Size of the problem along the z-axis (cm).
25    pub lz: T,
26
27    /// Size of a mesh cell along the x-axis (cm).
28    pub dx: T,
29    /// Size of a mesh cell along the y-axis (cm).
30    pub dy: T,
31    /// Size of a mesh cell along the z-axis (cm).
32    pub dz: T,
33}
34
35impl<T: CustomFloat> GlobalFccGrid<T> {
36    /// Constructor.
37    pub fn new(nx: usize, ny: usize, nz: usize, lx: T, ly: T, lz: T) -> Self {
38        let tmpx: T = FromPrimitive::from_usize(nx).unwrap();
39        let tmpy: T = FromPrimitive::from_usize(ny).unwrap();
40        let tmpz: T = FromPrimitive::from_usize(nz).unwrap();
41
42        Self {
43            nx,
44            ny,
45            nz,
46            lx,
47            ly,
48            lz,
49            dx: lx / tmpx,
50            dy: ly / tmpy,
51            dz: lz / tmpz,
52        }
53    }
54
55    /// Returns the index of the cell the coordinates belong to.
56    pub fn which_cell(&self, r: &MCVector<T>) -> usize {
57        let ix = r.x / self.dx;
58        let iy = r.y / self.dy;
59        let iz = r.z / self.dz;
60        self.cell_tuple_to_idx(&(
61            ix.to_usize().unwrap(),
62            iy.to_usize().unwrap(),
63            iz.to_usize().unwrap(),
64        ))
65    }
66
67    /// Returns the coordinate of the center of the given cell.
68    pub fn cell_center(&self, idx_cell: usize) -> MCVector<T> {
69        let two: T = FromPrimitive::from_f64(2.0).unwrap();
70        let tt: Tuple3 = self.cell_idx_to_tuple(idx_cell);
71        let r: MCVector<T> = self.node_coord_from_tuple(&(tt.0, tt.1, tt.2, 0));
72        r + MCVector {
73            x: self.dx / two,
74            y: self.dy / two,
75            z: self.dz / two,
76        }
77    }
78
79    /// Converts a cell index to a coordinate tuple.
80    pub fn cell_idx_to_tuple(&self, idx_cell: usize) -> Tuple3 {
81        let x = idx_cell % self.nx;
82        let tmp = idx_cell / self.nx;
83        let y = tmp % self.ny;
84        let z = tmp / self.ny;
85        (x, y, z)
86    }
87
88    /// Converts a cell coordinate tuple to an index.
89    pub fn cell_tuple_to_idx(&self, tuple_cell: &Tuple3) -> usize {
90        tuple_cell.0 + self.nx * (tuple_cell.1 + self.ny * tuple_cell.2)
91    }
92
93    /// Converts a node coordinate tuple to an index.
94    pub fn node_tuple_to_idx(&self, tt: &Tuple4) -> usize {
95        // nx, ny, nz are init with a +1 value in original code;
96        // nb is init at 4 but unused
97        tt.0 + (self.nx + 1) * (tt.1 + (self.ny + 1) * (tt.2 + (self.nz + 1) * tt.3))
98    }
99
100    /// Converts a node index to a coordinate tuple.
101    pub fn node_idx_to_tuple(&self, idx: usize) -> Tuple4 {
102        // nx, ny, nz are init with a +1 value in original code;
103        // nb is init at 4 but unused
104        let x = idx % (self.nx + 1);
105        let qx = idx / (self.nx + 1);
106        let y = qx % (self.ny + 1);
107        let qy = qx / (self.ny + 1);
108        let z = qy % (self.nz + 1);
109        let b = qy / (self.nz + 1);
110        (x, y, z, b)
111    }
112
113    /// Returns the global identifiers of the nodes of the specified cell.
114    pub fn get_node_gids(&self, cell_gid: usize) -> [usize; N_POINTS_INTERSEC] {
115        let mut node_gid: [usize; N_POINTS_INTERSEC] = [0; N_POINTS_INTERSEC];
116
117        let tt: Tuple3 = self.cell_idx_to_tuple(cell_gid);
118
119        (0..N_POINTS_INTERSEC).for_each(|ii| {
120            let tmp: Tuple4 = (
121                tt.0 + CORNER_OFFSET[ii].0,
122                tt.1 + CORNER_OFFSET[ii].1,
123                tt.2 + CORNER_OFFSET[ii].2,
124                CORNER_OFFSET[ii].3,
125            );
126            node_gid[ii] = self.node_tuple_to_idx(&tmp);
127        });
128
129        node_gid
130    }
131
132    /// Returns the global identifiers of the faces of the specified cell.
133    pub fn get_face_nbr_gids(&self, cell_gid: usize) -> [usize; N_FACES] {
134        let mut nbr_cell_gid: [usize; N_FACES] = [0; N_FACES];
135
136        let cell_tt = self.cell_idx_to_tuple(cell_gid);
137
138        (0..N_FACES).for_each(|ii| {
139            let face_nbr = (
140                cell_tt.0 as i32 + FACE_OFFSET[ii].0,
141                cell_tt.1 as i32 + FACE_OFFSET[ii].1,
142                cell_tt.2 as i32 + FACE_OFFSET[ii].2,
143            );
144            let snaped_face_nbr = self.snap_turtle(face_nbr);
145            nbr_cell_gid[ii] = self.cell_tuple_to_idx(&snaped_face_nbr);
146        });
147
148        nbr_cell_gid
149    }
150
151    /// Returns a node's coordinate from its index.
152    pub fn node_coord_from_idx(&self, idx: usize) -> MCVector<T> {
153        self.node_coord_from_tuple(&self.node_idx_to_tuple(idx))
154    }
155
156    /// Returns a node's coordinate from its tuple.
157    pub fn node_coord_from_tuple(&self, tt: &Tuple4) -> MCVector<T> {
158        let two: T = FromPrimitive::from_f64(2.0).unwrap();
159        let basis_offset: [MCVector<T>; 4] = [
160            MCVector::default(),
161            MCVector {
162                x: zero(),
163                y: self.dy / two,
164                z: self.dz / two,
165            },
166            MCVector {
167                x: self.dx / two,
168                y: zero(),
169                z: self.dz / two,
170            },
171            MCVector {
172                x: self.dx / two,
173                y: self.dy / two,
174                z: zero(),
175            },
176        ];
177        let tx: T = FromPrimitive::from_usize(tt.0).unwrap();
178        let ty: T = FromPrimitive::from_usize(tt.1).unwrap();
179        let tz: T = FromPrimitive::from_usize(tt.2).unwrap();
180
181        MCVector {
182            x: tx * self.dx,
183            y: ty * self.dy,
184            z: tz * self.dz,
185        } + basis_offset[tt.3]
186    }
187
188    /// Adjust the tuple value according to bounds.
189    pub fn snap_turtle(&self, tt: (i32, i32, i32)) -> Tuple3 {
190        // set tt such that (0 <= tt.* < n*)
191        (
192            tt.0.clamp(0, (self.nx - 1) as i32) as usize,
193            tt.1.clamp(0, (self.ny - 1) as i32) as usize,
194            tt.2.clamp(0, (self.nz - 1) as i32) as usize,
195        )
196    }
197}

203#[cfg(test)]
204mod tests {
205    use super::GlobalFccGrid;
206
207    #[test]
208    fn snap_turtle() {
209        let grid = GlobalFccGrid::new(3, 3, 3, 9.0, 9.0, 9.0);
210        let t0: (i32, i32, i32) = (0, 2, 1); // in bounds
211        let t1: (i32, i32, i32) = (3, -1, -2); // out of bounds
212        assert_eq!(grid.snap_turtle(t0), (0, 2, 1));
213        assert_eq!(grid.snap_turtle(t1), (2, 0, 0));
214    }
